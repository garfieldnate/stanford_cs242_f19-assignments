\begin{alignat*}{1}

\\
\msf{Key{\mhyphen}Value\ List}~kvl ::= \qamp kv \\
\mid \qamp kv , kvl \\
\\
\msf{Key{\mhyphen}Value}~kv ::= s : \tau
